### –û–ø–∏—Å–∞–Ω–∏–µ –∑–∞–¥–∞–Ω–∏—è

–ü—Ä–æ–≥—Ä–∞–º–º–∞ —Ä–µ–∞–ª–∏–∑—É–µ—Ç –≤–≤–æ–¥ –¥–∞–Ω–Ω—ã—Ö –æ —Å—Ç—É–¥–µ–Ω—Ç–∞—Ö (–ò–º—è, –ì—Ä—É–ø–ø–∞, GPA), —Å–æ—Ö—Ä–∞–Ω—è–µ—Ç –∏—Ö –≤ —Ç–µ–∫—Å—Ç–æ–≤—ã–π —Ñ–∞–π–ª (students.txt) –∏ —Å–µ—Ä–∏–∞–ª–∏–∑—É–µ—Ç –≤ —Ñ–æ—Ä–º–∞—Ç JSON (students.json). –¢–∞–∫–∂–µ —Ä–µ–∞–ª–∏–∑–æ–≤–∞–Ω–æ —á—Ç–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö –∏–∑ TXT-—Ñ–∞–π–ª–∞ –∏ –∏—Ö —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏—è –ø–æ –∑–∞–¥–∞–Ω–Ω–æ–º—É –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–º—É GPA. –í –∫–æ–¥ –¥–æ–±–∞–≤–ª–µ–Ω—ã —ç–ª–µ–º–µ–Ω—Ç—ã —É—Å–ª–æ–∂–Ω–µ–Ω–∏—è: –æ–±—Ä–∞–±–æ—Ç–∫–∞ –æ—à–∏–±–æ–∫ –≤–≤–æ–¥–∞ (try/except) –∏ —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –æ—Ç—Ñ–∏–ª—å—Ç—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞.

## 2. –ò—Å—Ö–æ–¥–Ω—ã–π –∫–æ–¥ –ø—Ä–æ–≥—Ä–∞–º–º—ã (main.py)
```python
import json

# --- –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –¥–∞–Ω–Ω—ã—Ö ---
students = []

# --- –í–≤–æ–¥ –¥–∞–Ω–Ω—ã—Ö –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–º ---
print("--- –í–≤–æ–¥ –¥–∞–Ω–Ω—ã—Ö ---")
try:
    n = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—É–¥–µ–Ω—Ç–æ–≤: "))
except ValueError:
    print("–û—à–∏–±–∫–∞: –í–≤–µ–¥–µ–Ω–æ –Ω–µ–≤–µ—Ä–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—É–¥–µ–Ω—Ç–æ–≤. –£—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–æ 0.")
    n = 0

for i in range(n):
    print(f"\n--- –°—Ç—É–¥–µ–Ω—Ç {i+1} ---")
    name = input("–ò–º—è —Å—Ç—É–¥–µ–Ω—Ç–∞: ")
    group = input("–ì—Ä—É–ø–ø–∞: ")
    
    gpa_valid = False
    while not gpa_valid:
        try:
            gpa = float(input("–°—Ä–µ–¥–Ω–∏–π –±–∞–ª–ª (GPA): "))
            if 0.0 <= gpa <= 5.0:
                gpa_valid = True
            else:
                print("GPA –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –≤ –¥–∏–∞–ø–∞–∑–æ–Ω–µ –æ—Ç 0.0 –¥–æ 5.0")
        except ValueError:
            print("–û—à–∏–±–∫–∞: –í–≤–µ–¥–∏—Ç–µ GPA –∫–∞–∫ —á–∏—Å–ª–æ (–Ω–∞–ø—Ä–∏–º–µ—Ä, 3.75).")
            
    # –î–æ–±–∞–≤–ª—è–µ–º –¥–∞–Ω–Ω—ã–µ –≤ —Å–ø–∏—Å–æ–∫
    students.append({"name": name, "group": group, "gpa": gpa})

# --- –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –≤ —Ç–µ–∫—Å—Ç–æ–≤—ã–π —Ñ–∞–π–ª (students.txt) ---
print("\n--- –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –≤ students.txt ---")
try:
    with open("students.txt", "w", encoding="utf-8") as f:
        for s in students:
            f.write(f"{s['name']} | {s['group']} | {s['gpa']}\n")
    print("–î–∞–Ω–Ω—ã–µ —É—Å–ø–µ—à–Ω–æ —Å–æ—Ö—Ä–∞–Ω–µ–Ω—ã –≤ students.txt.")
except IOError:
    print("–û—à–∏–±–∫–∞: –ù–µ —É–¥–∞–ª–æ—Å—å –∑–∞–ø–∏—Å–∞—Ç—å –¥–∞–Ω–Ω—ã–µ –≤ students.txt.")

# --- –ß—Ç–µ–Ω–∏–µ –∏ –≤—ã–≤–æ–¥ –¥–∞–Ω–Ω—ã—Ö –∏–∑ students.txt ---
print("\n--- –ß—Ç–µ–Ω–∏–µ –∏–∑ students.txt ---")
try:
    with open("students.txt", "r", encoding="utf-8") as f:
        print("–î–∞–Ω–Ω—ã–µ –∏–∑ —Ñ–∞–π–ª–∞ students.txt:")
        for line in f:
            print(line.strip())
except FileNotFoundError:
    print("–û—à–∏–±–∫–∞: –§–∞–π–ª students.txt –Ω–µ –Ω–∞–π–¥–µ–Ω.")

# --- –°–µ—Ä–∏–∞–ª–∏–∑–∞—Ü–∏—è –≤ JSON (students.json) ---
print("\n--- –°–µ—Ä–∏–∞–ª–∏–∑–∞—Ü–∏—è –≤ students.json ---")
try:
    with open("students.json", "w", encoding="utf-8") as jf:
        json.dump(students, jf, ensure_ascii=False, indent=4) 
    print("–î–∞–Ω–Ω—ã–µ —É—Å–ø–µ—à–Ω–æ —Å–µ—Ä–∏–∞–ª–∏–∑–æ–≤–∞–Ω—ã –≤ students.json.")
except IOError:
    print("–û—à–∏–±–∫–∞: –ù–µ —É–¥–∞–ª–æ—Å—å –∑–∞–ø–∏—Å–∞—Ç—å –¥–∞–Ω–Ω—ã–µ –≤ students.json.")

# --- –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è –ø–æ GPA ---
print("\n--- –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è –¥–∞–Ω–Ω—ã—Ö ---")
if students:
    try:
        threshold = float(input("–í–≤–µ–¥–∏—Ç–µ –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–π GPA –¥–ª—è —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏: "))
        
        filtered = [s for s in students if s["gpa"] >= threshold]

        print("\n–°—Ç—É–¥–µ–Ω—Ç—ã —Å GPA –≤—ã—à–µ –ø–æ—Ä–æ–≥–∞:")
        if filtered:
            filtered.sort(key=lambda s: s['gpa'], reverse=True)
            for s in filtered:
                print(f"{s['name']} ({s['group']}) ‚Äî GPA: {s['gpa']}")
        else:
            print("–°—Ç—É–¥–µ–Ω—Ç–æ–≤, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏—Ö –ø–æ—Ä–æ–≥—É, –Ω–µ –Ω–∞–π–¥–µ–Ω–æ.")
            
    except ValueError:
        print("–û—à–∏–±–∫–∞ —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏: –í–≤–µ–¥–∏—Ç–µ –ø–æ—Ä–æ–≥ GPA –∫–∞–∫ —á–∏—Å–ª–æ.")
else:
    print("–°–ø–∏—Å–æ–∫ —Å—Ç—É–¥–µ–Ω—Ç–æ–≤ –ø—É—Å—Ç, —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏—è –Ω–µ–≤–æ–∑–º–æ–∂–Ω–∞.")
```
